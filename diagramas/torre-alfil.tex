  \foreach\col in {2,4,...,8}

\foreach\row in {1,...,8}
  \node[anchor=east]  at (.4, \row) {\footnotesize\row};
\foreach\col in {1,...,8}
  \node[anchor=south] at (\col, .4) {\footnotesize\col};
}
