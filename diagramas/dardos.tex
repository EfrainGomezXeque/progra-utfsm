\documentclass[dvipsnames]{minimal}
\usepackage[pdftex,active,tightpage]{preview}
\usepackage{mathpazo}
\usepackage{tikz}

\tikzstyle{wired}=[draw=gray!30]
\tikzstyle{number}=[anchor=center, color=white]

% Sectors are numbered 0-19 counterclockwise from the top.

% \strip{color}{sector}{outer_radius}{inner_radius}
\newcommand{\strip}[4]{
    \filldraw[#1, wired]
      ({18 *  #2}      :                   #3) arc
      ({18 *  #2}      : {18 * (#2 + 1)} : #3) --
      ({18 * (#2 + 1)} :                   #4) arc
      ({18 * (#2 + 1)} : {18 *  #2}      : #4) -- cycle;
}

% \sector{color}{sector}{radius}
\newcommand{\sector}[3]{
    \filldraw[#1, wired]
      (0, 0) --
      ({18 * #2} :                   #3) arc
      ({18 * #2} : {18 * (#2 + 1)} : #3) -- cycle;
}

\begin{document}
\begin{preview}

% 81 degrees = 4.5 sectors.
% The rotation leaves 20 at the top.
\begin{tikzpicture}[rotate=81, scale=.25]

  % The whole board's background.
  \fill[black] (0, 0) circle (15cm);

  % Even sections.
  \foreach\i in {0,2,...,18} {
    \sector{black}{\i}{12cm}
    \strip{Red}{\i}{12cm}{11.5cm}
    \strip{Red}{\i}{ 7cm}{ 6.5cm}
  }

  % Odd sections.
  \foreach\i in {1,3,...,19} {
    \sector{white}{\i}{12cm}
    \strip{Green}{\i}{12cm}{11.5cm}
    \strip{Green}{\i}{ 7cm}{ 6.5cm}
  }

  % Bull's ring and eye.
  \filldraw[Green, wired] (0, 0) circle (0.9cm);
  \filldraw[Red,   wired] (0, 0) circle (0.4cm);

  % Labels.
  \foreach \sector/\label in {%
      0/20,  1/ 1,  2/18,  3/ 4,  4/13,
      5/ 6,  6/10,  7/15,  8/ 2,  9/17,
     10/ 3, 11/19, 12/ 7, 13/16, 14/ 8,
     15/11, 16/14, 17/ 9, 18/12, 19/ 5}%
  {
    \node[number] at ({18 * (-\sector + .5)} : 13.4cm) {\label};
  }

\end{tikzpicture}
\end{preview}

\end{document}
